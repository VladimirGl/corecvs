00169     {

00176                 ("Invalid coordinate in AbstractContiniousBuffer::elementBl(double y=%lf, double x=%lf) buffer sizes is [%dx%d]",

00178 
00179         ElementType a = this->element(i    ,j    );
00180         ElementType b = this->element(i    ,j + 1);
00181         ElementType c = this->element(i + 1,j    );
00182         ElementType d = this->element(i + 1,j + 1);
00183 
00184         double k1 = x - j;
00185         double k2 = y - i;
00186 
00187         ElementType result =
00188              (a * (1 - k1) + k1 * b) * (1 - k2) +
00189              (c * (1 - k1) + k1 * d) *      k2;
00190         return result;
