00029 void getCorrectionForPoint(vector2Dd input, LensCorrectionParametres *params, vector2Dd *output)
00030 {
00031         double dx = input.x - params->center.x;
00032         double dy = input.y - params->center.y;
00033         double dxsq = dx * dx;
00034         double dysq = dy * dy;
00035         double dxdy = dx * dy;
00036 
00037         double rsq = dxsq + dysq;
00038         double radialCorrection = params->k1 * rsq +  params->k2 * rsq * rsq;
00039 
00040         double radialX = (double)dx * radialCorrection;
00041         double radialY = (double)dy * radialCorrection;
00042 
00043         double tangentX = 2 * params->p1 * dxdy + params->p2 * ( rsq + 2 * dxsq );
00044         double tangentY = params->p1 * (rsq + 2 * dysq) + 2 * params->p2 * dxdy;
00045 
00046         output->x = radialX + tangentX;
00047         output->y = radialY + tangentY;
00048 }
