00073      * blockSAD8(): get UInt16x8 vector
00074      * \code
00075      *    [SAD(left,right) SAD(left+1,right) ... SAD(left+7,right) ]
00076      * \endcode

00080      *             left              left + 5
00081      *         +-----------+       +-----------+

00083      *         |   |   |   |       |   |   |   |
00084      *         |___|___|___|   ->  |___|___|___|

00091     static void blockSAD8(PuzzleBlock &left, PuzzleBlock &right, uint16_t* result)
00092     {
00093         UInt16x8 cost((uint16_t)0);
00094 
00095         uint16_t *dataL = NULL;
00096         uint16_t *dataR = NULL;
00097 
00098         for (int i = 0; i < left.h; i++)
00099         {
00100             dataL = left.data  + i * left.stride;
00101             dataR = right.data + i * right.stride;
00102 
00103             for (int j = 0; j < left.w; j++)
00104             {
00105                 UInt16x8 valL(dataL);
00106                 UInt16x8 valR(dataR[0]);
00107                 UInt16x8 diff = difference(valL, valR);
00108                 cost += diff;
00109                 dataL++;
00110                 dataR++;
00111             }
00112         }
00113         cost.save(result);
