00026 /**
00027  * \brief This function returns an inverse of the matrix.
00028  *
00029  * The calculation is done straight forward using the following formular property
00030  *
00031  *  \f[ I^{-1} = \frac 1 {|I|} \pmatrix{ M_{1,1} && \cdots } \f]
00032  *
00033  *
00034  * \attention No check is done, if the matrix is invertable or not
00035  *
00036  *
00037  *
00038  * \return The \f$I^{-1}\f$
00039  */
00040 Matrix33 Matrix33::inv()
00041 {
00042     Matrix33 toReturn;
00043     /* Will hold the minors determinant with the changing sign*/
00044     double ma =   a(1,1) * a(2,2) - a(1,2) * a(2,1);
00045     double mb = - a(1,0) * a(2,2) + a(1,2) * a(2,0);
00046     double mc =   a(1,0) * a(2,1) - a(1,1) * a(2,0);
00047 
00048     double det =   a(0,0) * ma
00049                  + a(0,1) * mb
00050                  + a(0,2) * mc;
00051 
00052     return Matrix33(
00053         ma / det,
00054         (a(0,2) * a(2,1) - a(0,1) * a(2,2) ) / det,
00055         (a(0,1) * a(1,2) - a(0,2) * a(1,1) ) / det,
00056 
00057         mb / det,
00058         (a(0,0) * a(2,2) - a(0,2) * a(2,0) ) / det,
00059         (a(0,2) * a(1,0) - a(0,0) * a(1,2) ) / det,
00060 
00061         mc / det,
00062         (a(0,1) * a(2,0) - a(0,0) * a(2,1) ) / det,
00063         (a(0,0) * a(1,1) - a(0,1) * a(1,0) ) / det
00064     );
