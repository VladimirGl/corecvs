00876 void dec_makehuff(struct dec_hufftbl *hu, int * hufflen, unsigned char *huffvals)
00877 {
00878     int code, k, i, j, d, x, c, v;
00879     for (i = 0; i < (1 << DECBITS); i++)
00880         hu->llvals[i] = 0;
00881 
00882     /*
00883  * llvals layout:
00884  *
00885  * value v already known, run r, backup u bits:
00886  *  vvvvvvvvvvvvvvvv 0000 rrrr 1 uuuuuuu
00887  * value unknown, size b bits, run r, backup u bits:
00888  *  000000000000bbbb 0000 rrrr 0 uuuuuuu
00889  * value and size unknown:
00890  *  0000000000000000 0000 0000 0 0000000
00891  */
00892     code = 0;
00893     k = 0;
00894     for (i = 0; i < 16; i++, code <<= 1) {    /* sizes */
00895         hu->valptr[i] = k;
00896         for (j = 0; j < hufflen[i]; j++) {
00897             hu->vals[k] = *huffvals++;
00898             if (i < DECBITS) {
00899                 c = code << (DECBITS - 1 - i);
00900                 v = hu->vals[k] & 0x0f;    /* size */
00901                 for (d = 1 << (DECBITS - 1 - i); --d >= 0;) {
00902                     if (v + i < DECBITS) {    /* both fit in table */
00903                         x = d >> (DECBITS - 1 - v - i);
00904                         if (v && x < (1 << (v - 1)))
00905                             x += (-1 << v) + 1;
00906                         x = x << 16 | (hu->vals[k] & 0xf0) << 4 |
00907                             (DECBITS - (i + 1 + v)) | 128;
00908                     } else
00909                         x = v << 16 | (hu->vals[k] & 0xf0) << 4 |
00910                             (DECBITS - (i + 1));
00911                     hu->llvals[c | d] = x;
00912                 }
00913             }
00914             code++;
00915             k++;
00916         }
00917         hu->maxcode[i] = code;
00918     }
00919     hu->maxcode[16] = 0x20000;    /* always terminate decode */
00920 }
