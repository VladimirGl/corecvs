00021 /**
00022  *
00023  *  This assignment algorithm uses the highly optimized version proposed by
00024  *  Daniel Shved
00025  *
00026  *  at http://acm.mipt.ru/twiki/bin/view/Algorithms/HungarianAlgorithmCPP
00027  *
00028  *  First of all - this version uses the optimization that allows not to store the
00029  *  duplicate of the matrix.
00030  *
00031  *  All assignment manipulations are limited to adding or subtracting
00032  *  value to of from column or row in whole. In this version the subtraction
00033  *  values are stored in the v and u arrays.
00034  *
00035  *
00036  *  Secondary - this algorithm processes matrix row by row.
00037  *
00038  *  If naive approach for assigning element in the newly added row doesn't work,
00039  *  (the column with the smallest element is already used in the assignment) *
00040  *  the algorithm analyzes the upper rows (markedI holds the currently analyzed row)
00041  *  until it will find a new collision-free assignment.
00042  *
00043  *
00044  * */
00045 
00046 template <typename BufferType, typename MappingType>
00047 bool assignmentOptimal(
00048         const BufferType *A,
00049         MappingType* mapping)
00050 {
00051    typedef typename BufferType::InternalElementType Type;
00052 
00053    int i, j, j1;
00054    int h = A->h;
00055    int w = A->w;
00056 
00057    Type   *v = new Type[w];
00058    Type   *u = new Type[h];
00059 
00060    int    *xy = new int[w];
00061 
00062    int   *links = new int[w];
00063    Type  *mins  = new Type[w];
00064    bool *visited  = new bool[w];
00065 
00066    for (i = 0; i < h; i++)
00067    {
00068        u[i] = Type(0);
00069    }
00070 
00071    for (j = 0; j < w; j++)
00072    {
00073        v[j] = Type(0);
00074        xy[j] = -1;
00075    }
00076 
00077 
00078    for(i = 0; i < h; i++) {
00079 
00080        /**
00081         * Staring new row analysis
00082         * While attempting to resolve collision we could build a chain of zeros
00083         * links will hold the connection between the rows
00084         * mins
00085         * visited - will hold the flags that the row is already visited
00086         *
00087         * */
00088        for (j = 0; j < w; j++)
00089        {
00090            links[j]   = -1;
00091            mins[j]    = std::numeric_limits<Type>::max();
00092            visited[j] = false;
00093        }
00094 
00095       /**
00096        * markedI holds the currently analyzed row.
00097        * Analysis starts with the just added row
00098        *
00099        * We will select  3 areas -
00100        * <ol>
00101        *   <li> unvisied rows
00102        *   <li> visied rows and visited columns - the visited block
00103        *   <li> visied rows and un visited columns - the search block
00104        * </ol>
00105        *
00106        *
00107        * */
00108       int markedI = i;
00109       int markedJ = -1;
00110       while(markedI != -1)
00111       {
00112           /**
00113            * Now we would update the information about the minimum in the un-visited columns
00114            * mins[j] will hold the the minimum value in the j-th column among all the
00115            * visited rows. Values in the visited columns do not matter.
00116            *
00117            * (j,markedI) will hold the position of the minimum in the search block
00118            *
00119            *
00120            * */
00121          j = -1;
00122          for(j1 = 0; j1 < w; j1 ++)
00123          {
00124             if(!visited[j1])
00125             {
00126                 Type current = A->element(markedI,j1) - u[markedI] - v[j1];
00127                 if(current < mins[j1]) {
00128                     /**
00129                      * We arrive here if the processed row in the j1 column holds the smallest element
00130                      * We update the mins array
00131                      * And create a horizontal link to the collision element
00132                      * */
00133                     mins[j1] = current;
00134                     links[j1] = markedJ;
00135                 }
00136                 if(j == -1 || mins[j1] < mins[j])
00137                    j = j1;
00138             }
00139          }
00140 
00141 
00142         /**
00143           * Now j hold the number of the unvisited column in which the minimum among the
00144           * visited rows is achieved.
00145           *
00146           * Thus delta will hold the minimum in the searchBlock.
00147           *
00148           * Next code will -
00149           *    In search block (visited rows, unvisited columns )
00150           *               decrease all the search block by delta - a new 0 will appear at (j, markedI)
00151           *
00152           *    In visited block (visited rows, visited columns)
00153           *               leave all unchanged (+ delta to columns, -delta to rows)
00154           *
00155           *    In the visited columns, unvisited rows
00156           *                  elements will increase. But there are no selected elements there
00157           *
00158           *    In the unvisited columns, unvisied rows
00159           *                nothing will happen
00160           *
00161           *
00162           * */
00163          Type delta = mins[j];
00164          for(j1 = 0; j1 < w; j1++)
00165          {
00166             if(visited[j1]) {
00167                u[xy[j1]] += delta;
00168                v[j1] -= delta;
00169             } else {
00170                mins[j1] -= delta;
00171             }
00172          }
00173          u[i] += delta;
00174 
00175          /**
00176           *  As said above (j, markedI) holds the new zero.  Now we have two options
00177           *  either xy[j] will hold -1 (which means no selected ("starred") zeros in the column j)
00178           *        then we will leave the cycle
00179           *  or there is another 0 in the row xy[j]
00180           *         then we make it current and continue to build links list.
00181           *        The new collision element is (j, xy[j])
00182           *
00183           * */
00184          visited[j] = 1;
00185          markedJ = j;
00186          markedI = xy[j];
00187       }
00188 
00189       /**
00190        *  We exited the collision processing cycle and now we have a linked list of collisions
00191        *
00192        *  Links holds the collision columns numbers  for "primed elements". It starts with the element in column j.
00193        *
00194        *  xy holds the map from columns to rows of the old assignment - the "stared elements"
00195        *
00196        *  We can now improve the list increasing it's size by new element and
00197        *  switching all the "stared" elements to "primed"
00198        *
00199        *  */
00200       for(; links[j] != -1; j = links[j])
00201          xy[j] = xy[links[j]];
00202       xy[j] = i;
00203    }
00204 
00205    // Returning the result
00206    for(j = 0; j < w; j++)
00207    {
00208       if(xy[j] != -1)
00209          mapping->element(xy[j],0) = j;
00210    }
00211 
00212    delete[] v;
00213    delete[] u;
00214    delete[] xy;
00215    delete[] links;
00216    delete[] mins;
00217    delete[] visited;
00218 
00219    return true;
00220 }
